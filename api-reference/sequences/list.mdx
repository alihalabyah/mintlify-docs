---
title: 'List Sequences'
api: 'GET /sequences'
description: 'List all sequences for the authenticated user with statistics'
---

## Query Parameters

<ParamField query="status" type="string">
  Filter by status: `draft`, `active`, `paused`, `archived`
</ParamField>

## Response

<ResponseField name="data" type="array">
  Array of sequence objects
  
  <Expandable title="Sequence Object">
    <ResponseField name="id" type="integer" required>
      Sequence ID
    </ResponseField>
    
    <ResponseField name="user_id" type="integer" required>
      Owner user ID
    </ResponseField>
    
    <ResponseField name="name" type="string" required>
      Sequence name (max 200 characters)
    </ResponseField>
    
    <ResponseField name="description" type="string">
      Sequence description
    </ResponseField>
    
    <ResponseField name="status" type="string" required>
      Current status: `draft`, `active`, `paused`, `archived`
    </ResponseField>
    
    <ResponseField name="instructions" type="string" required>
      Free-text playbook with AI instructions
    </ResponseField>
    
    <ResponseField name="email_templates" type="array">
      Array of email template objects (JSON)
    </ResponseField>
    
    <ResponseField name="total_leads" type="integer">
      Total number of enrolled leads (computed)
    </ResponseField>
    
    <ResponseField name="open_rate" type="number">
      Email open rate (0-1, computed)
    </ResponseField>
    
    <ResponseField name="reply_rate" type="number">
      Email reply rate (0-1, computed)
    </ResponseField>
    
    <ResponseField name="created_at" type="string" required>
      Creation timestamp
    </ResponseField>
    
    <ResponseField name="updated_at" type="string" required>
      Last update timestamp
    </ResponseField>
  </Expandable>
</ResponseField>
