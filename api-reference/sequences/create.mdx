---
title: 'Create Sequence'
api: 'POST /sequences'
description: 'Create a new sequence with instructions and email templates'
---

## Body Parameters

<ParamField body="name" type="string" required>
  Sequence name (max 200 characters)
</ParamField>

<ParamField body="description" type="string">
  Optional description of the sequence
</ParamField>

<ParamField body="instructions" type="string" required>
  Free-text playbook with high-level instructions for the AI (e.g., "ask introducer, then follow up...")
</ParamField>

<ParamField body="email_templates" type="array" required>
  Array of email template objects
  
  <Expandable title="Template Object">
    <ParamField body="name" type="string" required>
      Template name (e.g., "Step 1 - Intro")
    </ParamField>
    
    <ParamField body="subject" type="string" required>
      Email subject line
    </ParamField>
    
    <ParamField body="body" type="string" required>
      Email body (HTML)
    </ParamField>
  </Expandable>
</ParamField>

<RequestExample>
```json
{
  "name": "Outbound – Cold V1",
  "description": "Initial outreach sequence for cold leads",
  "instructions": "Start with introducer mention, highlight mutual connection value, ask for intro call. Follow up after 3 days if no response.",
  "email_templates": [
    {
      "name": "Step 1 - Intro",
      "subject": "Quick intro via {{introducer_name}}",
      "body": "<p>Hi {{lead_name}},</p><p>{{introducer_name}} suggested I reach out...</p>"
    },
    {
      "name": "Step 2 - Follow-up",
      "subject": "Re: Quick intro",
      "body": "<p>Following up on my previous email...</p>"
    }
  ]
}
```
</RequestExample>

## Response

<ResponseField name="data" type="object">
  Created sequence object
  
  <Expandable title="Sequence Object">
    <ResponseField name="id" type="integer" required>
      Sequence ID
    </ResponseField>
    
    <ResponseField name="user_id" type="integer" required>
      Owner user ID
    </ResponseField>
    
    <ResponseField name="name" type="string" required>
      Sequence name
    </ResponseField>
    
    <ResponseField name="description" type="string">
      Sequence description
    </ResponseField>
    
    <ResponseField name="status" type="string" required>
      Initial status (always `draft`)
    </ResponseField>
    
    <ResponseField name="instructions" type="string" required>
      AI instructions playbook
    </ResponseField>
    
    <ResponseField name="email_templates" type="array" required>
      Email templates array
    </ResponseField>
    
    <ResponseField name="created_at" type="string" required>
      Creation timestamp (ISO 8601)
    </ResponseField>
    
    <ResponseField name="updated_at" type="string" required>
      Last update timestamp (ISO 8601)
    </ResponseField>
  </Expandable>
</ResponseField>
