---
title: 'Create Sequence'
api: 'POST /sequences'
description: 'Create a new sequence with instructions, templates, and context'
---

## Body Parameters

<ParamField body="name" type="string" required>
  Sequence name
</ParamField>

<ParamField body="instruction" type="string" required>
  High-level instructions for the AI agent
</ParamField>

<ParamField body="lead_context" type="string" required>
  Context about the user/sender (profile, experience, etc.)
</ParamField>

<ParamField body="context" type="string">
  Additional context for personalization
</ParamField>

<ParamField body="templates" type="array" required>
  Array of email template objects
  
  <Expandable title="Template Object">
    <ParamField body="name" type="string" required>
      Template name (e.g., "Step 1 - Intro")
    </ParamField>
    
    <ParamField body="subject" type="string" required>
      Email subject line
    </ParamField>
    
    <ParamField body="body" type="string" required>
      Email body (HTML)
    </ParamField>
  </Expandable>
</ParamField>

<ParamField body="notes" type="string">
  Internal notes
</ParamField>

<RequestExample>
```json
{
  "name": "Outbound – Cold V1",
  "instruction": "High-level rules...",
  "lead_context": "<your_profile> Name: Abdallah Absi Work Experience: 1. CEO at Village (village.ai), Jan 2022 - Present",
  "context": "...",
  "templates": [
    {
      "name": "Step 1 - Intro",
      "subject": "Quick intro",
      "body": "<p>...</p>"
    }
  ],
  "notes": "internal notes"
}
```
</RequestExample>

## Response

<ResponseField name="data" type="object">
  Created sequence object
  
  <Expandable title="Sequence Object">
    <ResponseField name="id" type="integer" required>
      Sequence ID
    </ResponseField>
    
    <ResponseField name="name" type="string" required>
      Sequence name
    </ResponseField>
    
    <ResponseField name="status" type="string" required>
      Initial status (always "draft")
    </ResponseField>
    
    <ResponseField name="instruction" type="string" required>
      AI instructions
    </ResponseField>
    
    <ResponseField name="lead_context" type="string" required>
      User context
    </ResponseField>
    
    <ResponseField name="context" type="string">
      Additional context
    </ResponseField>
    
    <ResponseField name="templates" type="array" required>
      Email templates
    </ResponseField>
    
    <ResponseField name="notes" type="string">
      Internal notes
    </ResponseField>
    
    <ResponseField name="created_at" type="string" required>
      Creation timestamp
    </ResponseField>
    
    <ResponseField name="updated_at" type="string" required>
      Last update timestamp
    </ResponseField>
  </Expandable>
</ResponseField>
