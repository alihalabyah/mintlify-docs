---
title: 'Get Sequence'
api: 'GET /sequences/:id'
description: 'Load a specific sequence with all configuration details'
---

## Path Parameters

<ParamField path="id" type="integer" required>
  Sequence ID
</ParamField>

## Response

<ResponseField name="data" type="object">
  Sequence object with full details
  
  <Expandable title="Sequence Object">
    <ResponseField name="id" type="integer" required>
      Sequence ID
    </ResponseField>
    
    <ResponseField name="user_id" type="integer" required>
      Owner user ID
    </ResponseField>
    
    <ResponseField name="name" type="string" required>
      Sequence name
    </ResponseField>
    
    <ResponseField name="description" type="string">
      Sequence description
    </ResponseField>
    
    <ResponseField name="status" type="string" required>
      Current status: `draft`, `active`, `paused`, `archived`
    </ResponseField>
    
    <ResponseField name="instructions" type="string" required>
      AI instructions playbook
    </ResponseField>
    
    <ResponseField name="email_templates" type="array" required>
      Email templates
    </ResponseField>
    
    <ResponseField name="created_at" type="string" required>
      Creation timestamp
    </ResponseField>
    
    <ResponseField name="updated_at" type="string" required>
      Last update timestamp
    </ResponseField>
  </Expandable>
</ResponseField>
