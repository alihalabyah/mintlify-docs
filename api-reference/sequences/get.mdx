---
title: 'Get Sequence'
api: 'GET /sequences/:id'
description: 'Load a specific sequence with all configuration details'
---

## Path Parameters

<ParamField path="id" type="integer" required>
  Sequence ID
</ParamField>

## Response

<ResponseField name="data" type="object">
  Sequence object with full details
  
  <Expandable title="Sequence Object">
    <ResponseField name="id" type="integer" required>
      Sequence ID
    </ResponseField>
    
    <ResponseField name="name" type="string" required>
      Sequence name
    </ResponseField>
    
    <ResponseField name="status" type="string" required>
      Current status
    </ResponseField>
    
    <ResponseField name="lead_context" type="string" required>
      User/sender context
    </ResponseField>
    
    <ResponseField name="context" type="string">
      Additional context
    </ResponseField>
    
    <ResponseField name="instruction" type="string" required>
      AI instructions
    </ResponseField>
    
    <ResponseField name="templates" type="array" required>
      Email templates
    </ResponseField>
    
    <ResponseField name="notes" type="string">
      Internal notes
    </ResponseField>
    
    <ResponseField name="created_at" type="string" required>
      Creation timestamp
    </ResponseField>
    
    <ResponseField name="updated_at" type="string" required>
      Last update timestamp
    </ResponseField>
  </Expandable>
</ResponseField>
