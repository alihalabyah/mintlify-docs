---
title: 'Cancel Scheduled Email'
api: 'DELETE /sequences/:sequenceId/emails/:emailId'
description: 'Cancel a scheduled email by setting status to failed with cancellation metadata'
---

## Path Parameters

<ParamField path="sequenceId" type="integer" required>
  Sequence ID
</ParamField>

<ParamField path="emailId" type="integer" required>
  Email ID from email_outgoing table
</ParamField>

## Response

<ResponseField name="data" type="object">
  Canceled email object
  
  <Expandable title="Canceled Email Object">
    <ResponseField name="id" type="integer" required>
      Email ID
    </ResponseField>
    
    <ResponseField name="sequence_lead_id" type="integer" required>
      Sequence lead ID
    </ResponseField>
    
    <ResponseField name="status" type="string" required>
      Status (will be `failed`)
    </ResponseField>
    
    <ResponseField name="content" type="object" required>
      Email content (preserved)
    </ResponseField>
    
    <ResponseField name="due_at" type="string">
      Original scheduled time
    </ResponseField>
    
    <ResponseField name="updated_at" type="string" required>
      Cancellation timestamp
    </ResponseField>
  </Expandable>
</ResponseField>

</Note>
