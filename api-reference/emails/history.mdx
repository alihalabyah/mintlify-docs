---
title: 'Get Lead Email History'
api: 'GET /sequences/:sequenceId/leads/:leadId/emails'
description: 'Get complete email thread history (sent + received) for a specific lead'
---

## Path Parameters

<ParamField path="sequenceId" type="integer" required>
  Sequence ID
</ParamField>

<ParamField path="leadId" type="integer" required>
  Lead ID
</ParamField>

## Response

<ResponseField name="data" type="object">
  <Expandable title="Email History Object">
    <ResponseField name="lead" type="object" required>
      Lead information (lead_id, lead_name, email)
    </ResponseField>
    
    <ResponseField name="email_threads" type="array" required>
      Array of email thread objects
      
      <Expandable title="Thread Object">
        <ResponseField name="thread_id" type="string" required>
          Thread identifier
        </ResponseField>
        
        <ResponseField name="emails" type="array" required>
          Array of emails in the thread (both outgoing and inbox)
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>
