---
title: 'Get Lead Email History'
api: 'GET /sequences/:sequenceId/leads/:leadId/emails'
description: 'Get complete email history (outgoing + incoming) for a specific lead, organized by threads'
---

## Path Parameters

<ParamField path="sequenceId" type="integer" required>
  Sequence ID
</ParamField>

<ParamField path="leadId" type="integer" required>
  Lead ID
</ParamField>

## Response

<ResponseField name="data" type="object">
  <Expandable title="Email History Object">
    <ResponseField name="lead" type="object" required>
      Lead information
      
      <Expandable title="Lead Info">
        <ResponseField name="lead_id" type="integer" required>
          Lead ID
        </ResponseField>
        
        <ResponseField name="sequence_lead_id" type="integer" required>
          Sequence lead ID
        </ResponseField>
        
        <ResponseField name="lead_data" type="object" required>
          Lead information (name, email, company, etc.)
        </ResponseField>
      </Expandable>
    </ResponseField>
    
    <ResponseField name="email_threads" type="array" required>
      Array of email thread objects
      
      <Expandable title="Thread Object">
        <ResponseField name="thread_id" type="string" required>
          Gmail thread ID
        </ResponseField>
        
        <ResponseField name="emails" type="array" required>
          Array of emails in chronological order
          
          <Expandable title="Email Item">
            <ResponseField name="type" type="string" required>
              Email type: `outgoing` or `incoming`
            </ResponseField>
            
            <ResponseField name="id" type="integer" required>
              Email ID (from email_outgoing or email_inbox)
            </ResponseField>
            
            <ResponseField name="content" type="object">
              Email content (outgoing) or body preview (incoming)
            </ResponseField>
            
            <ResponseField name="from_email" type="string">
              Sender email (for incoming)
            </ResponseField>
            
            <ResponseField name="to_email" type="string">
              Recipient email (for outgoing)
            </ResponseField>
            
            <ResponseField name="sent_at" type="string">
              Send timestamp (outgoing)
            </ResponseField>
            
            <ResponseField name="received_at" type="string">
              Received timestamp (incoming)
            </ResponseField>
            
            <ResponseField name="status" type="string">
              Email status (outgoing)
            </ResponseField>
            
            <ResponseField name="type_detail" type="string">
              Incoming email type: `replied`, `oof`, `bounced`, etc.
            </ResponseField>
          </Expandable>
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>
