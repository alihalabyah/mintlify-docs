---
title: 'Get Email Engagement Activity'
api: 'GET /emails/:id/engagement'
description: 'Get engagement events and replies for a sent email (delivered, opened, clicked, replied, bounced)'
---

## Path Parameters

<ParamField path="id" type="integer" required>
  Email ID from email_outgoing table
</ParamField>

## Response

<ResponseField name="data" type="object">
  <Expandable title="Engagement Data">
    <ResponseField name="email" type="object" required>
      Email summary information
      
      <Expandable title="Email Info">
        <ResponseField name="id" type="integer" required>
          Email ID
        </ResponseField>
        
        <ResponseField name="integration_message_id" type="string">
          Gmail message ID
        </ResponseField>
        
        <ResponseField name="integration_thread_id" type="string">
          Gmail thread ID
        </ResponseField>
        
        <ResponseField name="sent_at" type="string">
          Send timestamp
        </ResponseField>
      </Expandable>
    </ResponseField>
    
    <ResponseField name="events" type="array" required>
      Array of engagement events from email_inbox (type: delivered, opened, clicked, bounced)
      
      <Expandable title="Event Object">
        <ResponseField name="id" type="integer" required>
          Event ID (email_inbox record)
        </ResponseField>
        
        <ResponseField name="type" type="string" required>
          Event type: `delivered`, `opened`, `clicked`, `bounced`, `other`
        </ResponseField>
        
        <ResponseField name="received_at" type="string" required>
          Event timestamp
        </ResponseField>
        
        <ResponseField name="content" type="object">
          Event metadata (user agent, click URL, etc.)
        </ResponseField>
      </Expandable>
    </ResponseField>
    
    <ResponseField name="replies" type="array" required>
      Array of reply emails from email_inbox (type: replied)
      
      <Expandable title="Reply Object">
        <ResponseField name="id" type="integer" required>
          Reply email ID
        </ResponseField>
        
        <ResponseField name="from_email" type="string" required>
          Sender email address
        </ResponseField>
        
        <ResponseField name="received_at" type="string" required>
          Receipt timestamp
        </ResponseField>
        
        <ResponseField name="content" type="object">
          Email content with subject and body preview
        </ResponseField>
        
        <ResponseField name="type" type="string" required>
          Will be `replied`
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

<Note>
  Engagement tracking relies on webhook processing from Gmail. Events are stored in the `email_inbox` table with the corresponding `integration_message_id` and categorized by type (delivered, opened, clicked, replied, bounced, oof, other).
</Note>
