---
title: 'Get Email Engagement Activity'
api: 'GET /emails/:id/activities'
description: 'Get engagement timeline for a sent email (delivered, opened, clicked, replied, bounced)'
---

## Path Parameters

<ParamField path="id" type="integer" required>
  Email ID from email_outgoing table
</ParamField>

## Response

<ResponseField name="data" type="object">
  <Expandable title="Engagement Data">
    <ResponseField name="events" type="array" required>
      Array of engagement event objects (delivered, opened, clicked, bounced)
      
      <Expandable title="Event Object">
        <ResponseField name="id" type="integer" required>
          Event ID
        </ResponseField>
        
        <ResponseField name="type" type="string" required>
          Event type: `delivered`, `opened`, `clicked`, `bounced`
        </ResponseField>
        
        <ResponseField name="created_at" type="string" required>
          Event timestamp
        </ResponseField>
        
        <ResponseField name="user_agent" type="string">
          User agent (for opens/clicks)
        </ResponseField>
      </Expandable>
    </ResponseField>
    
    <ResponseField name="replies" type="array" required>
      Array of reply email objects
      
      <Expandable title="Reply Object">
        <ResponseField name="id" type="integer" required>
          Reply email ID
        </ResponseField>
        
        <ResponseField name="from_email" type="string" required>
          Sender email address
        </ResponseField>
        
        <ResponseField name="received_at" type="string" required>
          Receipt timestamp
        </ResponseField>
        
        <ResponseField name="body_preview" type="string">
          Preview of reply body
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>
