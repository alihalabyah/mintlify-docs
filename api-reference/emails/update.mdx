---
title: 'Update Scheduled Email'
api: 'PATCH /emails/:id'
description: "Edit a draft or scheduled email's content or send time (only if not yet sent)"
---

## Path Parameters

<ParamField path="id" type="integer" required>
  Email ID from email_outgoing table
</ParamField>

## Body Parameters

<ParamField body="content" type="object">
  Updated email content
  
  <Expandable title="Content Object">
    <ParamField body="subject" type="string">
      Email subject line
    </ParamField>
    
    <ParamField body="body" type="string">
      Email body (HTML)
    </ParamField>
  </Expandable>
</ParamField>

<ParamField body="due_at" type="string">
  Updated send time (ISO 8601 timestamp)
</ParamField>

<ParamField body="to_email" type="string">
  Update recipient email address
</ParamField>

<ParamField body="cc_emails" type="array">
  Update CC recipients
</ParamField>

<ParamField body="reply_in_thread" type="boolean">
  Whether to reply in existing thread
</ParamField>

<RequestExample>
```json
{
  "content": {
    "subject": "Updated subject",
    "body": "<p>Updated content...</p>"
  },
  "due_at": "2025-11-12T09:30:00Z"
}
```
</RequestExample>

## Response

<ResponseField name="data" type="object">
  Updated email object
  
  <Expandable title="Email Object">
    <ResponseField name="id" type="integer" required>
      Email ID
    </ResponseField>
    
    <ResponseField name="sequence_lead_id" type="integer" required>
      Sequence lead ID
    </ResponseField>
    
    <ResponseField name="status" type="string" required>
      Email status
    </ResponseField>
    
    <ResponseField name="content" type="object" required>
      Updated email content
    </ResponseField>
    
    <ResponseField name="due_at" type="string">
      Updated scheduled time
    </ResponseField>
    
    <ResponseField name="to_email" type="string" required>
      Recipient email
    </ResponseField>
    
    <ResponseField name="updated_at" type="string" required>
      Last update timestamp
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseExample>
```json
{
  "data": {
    "id": 9002,
    "sequence_lead_id": 1001,
    "sequence_lead_activity_id": 321,
    "integration_id": 42,
    "status": "draft",
    "content": {
      "subject": "Updated subject",
      "body": "<p>Updated content...</p>"
    },
    "due_at": "2025-11-12T09:30:00Z",
    "scheduled_at": null,
    "sent_at": null,
    "to_email": "jane@acme.com",
    "reply_in_thread": true,
    "updated_at": "2025-11-11T15:00:00Z"
  },
  "metadata": {},
  "pagination": null
}
```
</ResponseExample>

<Note>
  Only emails with status `draft` or `scheduled` can be updated. Once an email reaches `sending` or `sent` status, it cannot be modified.
</Note>
