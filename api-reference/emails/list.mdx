---
title: 'List Sequence Emails'
api: 'GET /sequences/:id/emails'
description: 'List all outgoing emails (draft, scheduled, and sent) for a sequence'
---

## Path Parameters

<ParamField path="id" type="integer" required>
  Sequence ID
</ParamField>

## Query Parameters

<ParamField query="status" type="string">
  Filter by email status: `draft`, `scheduled`, `sending`, `sent`, `failed`
</ParamField>

<ParamField query="sequence_lead_id" type="integer">
  Filter by specific sequence lead
</ParamField>

## Response

<ResponseField name="data" type="array">
  Array of email_outgoing objects
  
  <Expandable title="Email Object">
    <ResponseField name="id" type="integer" required>
      Email ID
    </ResponseField>
    
    <ResponseField name="sequence_lead_id" type="integer" required>
      Sequence lead ID
    </ResponseField>
    
    <ResponseField name="sequence_lead_activity_id" type="integer" required>
      Associated activity ID
    </ResponseField>
    
    <ResponseField name="integration_id" type="integer" required>
      Mailbox integration ID used for sending
    </ResponseField>
    
    <ResponseField name="status" type="string" required>
      Email status: `draft`, `scheduled`, `sending`, `sent`, `failed`
    </ResponseField>
    
    <ResponseField name="due_at" type="string">
      When the email should be sent
    </ResponseField>
    
    <ResponseField name="scheduled_at" type="string">
      When the email was actually scheduled (after rate limiting)
    </ResponseField>
    
    <ResponseField name="sent_at" type="string">
      When the email was successfully sent
    </ResponseField>
    
    <ResponseField name="content" type="object" required>
      Email content (subject, body, etc.)
    </ResponseField>
    
    <ResponseField name="reply_in_thread" type="boolean" required>
      Whether to reply in existing thread
    </ResponseField>
    
    <ResponseField name="to_email" type="string" required>
      Recipient email address
    </ResponseField>
    
    <ResponseField name="cc_emails" type="array">
      CC recipients (JSON array)
    </ResponseField>
    
    <ResponseField name="integration_message_id" type="string">
      Gmail message ID (set after sending)
    </ResponseField>
    
    <ResponseField name="integration_thread_id" type="string">
      Gmail thread ID (set after sending)
    </ResponseField>
    
    <ResponseField name="created_at" type="string" required>
      Creation timestamp
    </ResponseField>
    
    <ResponseField name="updated_at" type="string" required>
      Last update timestamp
    </ResponseField>
  </Expandable>
</ResponseField>
