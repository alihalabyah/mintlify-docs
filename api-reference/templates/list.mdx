---
title: 'Get System Templates'
api: 'GET /templates'
description: 'Get system-provided email templates for the template library picker'
---

## Response

<ResponseField name="data" type="array">
  Array of template objects
  
  <Expandable title="Template Object">
    <ResponseField name="id" type="integer" required>
      Template ID
    </ResponseField>
    
    <ResponseField name="name" type="string" required>
      Template name
    </ResponseField>
    
    <ResponseField name="subject" type="string" required>
      Default subject line
    </ResponseField>
    
    <ResponseField name="body" type="string" required>
      Template body (HTML)
    </ResponseField>
    
    <ResponseField name="category" type="string" required>
      Template category (e.g., "follow_up", "intro")
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseExample>
```json
{
  "data": [
    {
      "id": 1,
      "name": "Follow-up",
      "subject": "Quick update",
      "body": "<p>...</p>",
      "category": "follow_up"
    }
  ],
  "metadata": {},
  "pagination": {
    "cursor": null,
    "has_more": false,
    "count": 1
  }
}
```
</ResponseExample>
