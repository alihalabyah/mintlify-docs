---
title: 'List Pending Activities'
api: 'GET /activities'
description: 'List pending activities awaiting user approval (typically outgoing email drafts)'
---

## Query Parameters

<ParamField query="activity_type" type="string">
  Filter by activity type: `incoming_email`, `outgoing_email`, `system_note`
</ParamField>

<ParamField query="status" type="string">
  Filter by status: `pending`, `approved`, `rejected`
</ParamField>

<ParamField query="sequence_id" type="integer">
  Filter by specific sequence
</ParamField>

## Response

<ResponseField name="data" type="array">
  Array of activity objects
  
  <Expandable title="Activity Object">
    <ResponseField name="id" type="integer" required>
      Activity ID
    </ResponseField>
    
    <ResponseField name="sequence_lead_id" type="integer" required>
      Sequence lead ID
    </ResponseField>
    
    <ResponseField name="activity_type" type="string" required>
      Activity type: `incoming_email`, `outgoing_email`, `system_note`
    </ResponseField>
    
    <ResponseField name="is_approved" type="boolean">
      Approval status (null for non-approvable activities)
    </ResponseField>
    
    <ResponseField name="status" type="string">
      Status: `pending`, `approved`, `rejected`
    </ResponseField>
    
    <ResponseField name="due_at" type="string">
      When the action is planned to execute
    </ResponseField>
    
    <ResponseField name="executed_at" type="string">
      When the action was actually executed
    </ResponseField>
    
    <ResponseField name="outgoing_email_id" type="integer">
      Reference to email_outgoing record
    </ResponseField>
    
    <ResponseField name="incoming_email_id" type="integer">
      Reference to email_inbox record
    </ResponseField>
    
    <ResponseField name="integration_thread_id" type="string">
      Gmail thread ID
    </ResponseField>
    
    <ResponseField name="integration_message_id" type="string">
      Gmail message ID
    </ResponseField>
    
    <ResponseField name="ai_confidence_score" type="number">
      AI confidence score (0.00-1.00)
    </ResponseField>
    
    <ResponseField name="ai_payload" type="object">
      Full AI context (step chosen, template used, reasoning, etc.)
    </ResponseField>
    
    <ResponseField name="rejection_reason" type="string">
      User-provided rejection reason
    </ResponseField>
    
    <ResponseField name="created_at" type="string" required>
      Creation timestamp
    </ResponseField>
  </Expandable>
</ResponseField>
