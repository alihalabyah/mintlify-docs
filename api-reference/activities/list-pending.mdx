---
title: 'List Pending Activities'
api: 'GET /activities'
description: 'List pending draft emails awaiting user approval'
---

## Query Parameters

<ParamField query="activity_type" type="string">
  Filter by activity type (e.g., `draft_email`)
</ParamField>

<ParamField query="status" type="string">
  Filter by status (e.g., `pending`)
</ParamField>

## Response

<ResponseField name="data" type="array">
  Array of pending activity objects
  
  <Expandable title="Activity Object">
    <ResponseField name="id" type="integer" required>
      Activity ID
    </ResponseField>
    
    <ResponseField name="sequence_lead_id" type="integer" required>
      Sequence lead ID
    </ResponseField>
    
    <ResponseField name="sequence_id" type="integer" required>
      Sequence ID
    </ResponseField>
    
    <ResponseField name="sequence_name" type="string" required>
      Sequence name
    </ResponseField>
    
    <ResponseField name="lead_name" type="string" required>
      Lead name
    </ResponseField>
    
    <ResponseField name="activity_type" type="string" required>
      Activity type
    </ResponseField>
    
    <ResponseField name="status" type="string" required>
      Current status
    </ResponseField>
    
    <ResponseField name="subject" type="string">
      Email subject
    </ResponseField>
    
    <ResponseField name="body_html" type="string">
      Email body (HTML)
    </ResponseField>
    
    <ResponseField name="due_at" type="string">
      Scheduled execution time
    </ResponseField>
    
    <ResponseField name="ai_confidence" type="number">
      AI confidence score (0-1)
    </ResponseField>
    
    <ResponseField name="ai_reason" type="string">
      AI reasoning for the draft
    </ResponseField>
    
    <ResponseField name="created_at" type="string" required>
      Creation timestamp
    </ResponseField>
  </Expandable>
</ResponseField>
