---
title: 'Approve Activity'
api: 'POST /sequences/:sequenceId/activities/:activityId/approve'
description: 'Approve an activity and schedule it for execution'
---

## Path Parameters

<ParamField path="sequenceId" type="integer" required>
  Sequence ID
</ParamField>

<ParamField path="activityId" type="integer" required>
  Activity ID
</ParamField>

## Body Parameters

<ParamField body="modifications" type="object">
  Optional modifications to apply before approval (will update the associated email_outgoing record)
  
  <Expandable title="Modification Options">
    <ParamField body="subject" type="string">
      Override email subject
    </ParamField>
    
    <ParamField body="body" type="string">
      Override email body (HTML)
    </ParamField>
    
    <ParamField body="due_at" type="string">
      Override scheduled send time (ISO 8601)
    </ParamField>
  </Expandable>
</ParamField>

<RequestExample>
```json
{
  "modifications": {
    "subject": "Final subject line",
    "body": "<p>Finalized email content...</p>",
    "due_at": "2025-11-11T09:00:00Z"
  }
}
```
</RequestExample>

## Response

<ResponseField name="data" type="object">
  <Expandable title="Approval Result">
    <ResponseField name="activity" type="object" required>
      Updated activity object
      
      <Expandable title="Activity Object">
        <ResponseField name="id" type="integer" required>
          Activity ID
        </ResponseField>
        
        <ResponseField name="status" type="string" required>
          Status (will be `approved`)
        </ResponseField>
        
        <ResponseField name="is_approved" type="boolean" required>
          Approval flag (will be true)
        </ResponseField>
        
        <ResponseField name="executed_at" type="string">
          Set when email is actually sent
        </ResponseField>
      </Expandable>
    </ResponseField>
    
    <ResponseField name="email" type="object" required>
      Associated email_outgoing record updated to scheduled status
      
      <Expandable title="Email Object">
        <ResponseField name="id" type="integer" required>
          Email ID
        </ResponseField>
        
        <ResponseField name="sequence_lead_id" type="integer" required>
          Sequence lead ID
        </ResponseField>
        
        <ResponseField name="status" type="string" required>
          Email status (will be `scheduled`)
        </ResponseField>
        
        <ResponseField name="content" type="object" required>
          Email content (subject, body)
        </ResponseField>
        
        <ResponseField name="due_at" type="string" required>
          Scheduled send time
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseExample>
```json
{
  "data": {
    "activity": {
      "id": 321,
      "sequence_lead_id": 1001,
      "status": "approved",
      "is_approved": true,
      "executed_at": null,
      "created_at": "2025-11-10T15:00:00Z"
    },
    "email": {
      "id": 9003,
      "sequence_lead_id": 1001,
      "sequence_lead_activity_id": 321,
      "integration_id": 42,
      "status": "scheduled",
      "content": {
        "subject": "Final subject line",
        "body": "<p>Finalized email content...</p>"
      },
      "due_at": "2025-11-11T09:00:00Z",

    }
  },
  "metadata": {},
  "pagination": null
}
```
</ResponseExample>

<Note>
  After approval, the email enters the sending queue and will be processed by the email sending workflow (Workflow #6) which applies rate limits and schedules actual delivery.
</Note>
