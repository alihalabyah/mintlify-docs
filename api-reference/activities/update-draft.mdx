---
title: 'Update Draft Activity'
api: 'PATCH /activities/:id'
description: 'Update the email content or scheduled time of a pending draft activity'
---

## Path Parameters

<ParamField path="id" type="integer" required>
  Activity ID
</ParamField>

## Body Parameters

<ParamField body="subject" type="string">
  Updated email subject
</ParamField>

<ParamField body="body" type="string">
  Updated email body (HTML)
</ParamField>

<ParamField body="due_at" type="string">
  Updated scheduled time (ISO 8601)
</ParamField>

<RequestExample>
```json
{
  "subject": "Updated subject",
  "body": "<p>Updated content...</p>",
  "due_at": "2025-11-11T10:00:00Z"
}
```
</RequestExample>

## Response

<ResponseField name="data" type="object">
  Updated activity and email objects
  
  <Expandable title="Update Result">
    <ResponseField name="activity" type="object" required>
      Updated activity object
    </ResponseField>
    
    <ResponseField name="email" type="object" required>
      Associated email_outgoing record with updated content
      
      <Expandable title="Email Object">
        <ResponseField name="id" type="integer" required>
          Email ID
        </ResponseField>
        
        <ResponseField name="content" type="object" required>
          Updated email content (subject, body)
        </ResponseField>
        
        <ResponseField name="due_at" type="string">
          Updated scheduled time
        </ResponseField>
        
        <ResponseField name="updated_at" type="string" required>
          Last update timestamp
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseExample>
```json
{
  "data": {
    "activity": {
      "id": 321,
      "sequence_lead_id": 1001,
      "activity_type": "outgoing_email",
      "status": "pending",
      "outgoing_email_id": 9003,
      "due_at": "2025-11-11T10:00:00Z"
    },
    "email": {
      "id": 9003,
      "sequence_lead_id": 1001,
      "status": "draft",
      "content": {
        "subject": "Updated subject",
        "body": "<p>Updated content...</p>"
      },
      "due_at": "2025-11-11T10:00:00Z",
      "updated_at": "2025-11-10T16:00:00Z"
    }
  },
  "metadata": {},
  "pagination": null
}
```
</ResponseExample>

<Note>
  Users can iterate on drafts multiple times before approving. Each update modifies the associated `email_outgoing` record while keeping the activity in `pending` status.
</Note>
