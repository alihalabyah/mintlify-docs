---
title: 'Reject Draft Email'
api: 'POST /activities/:id/reject'
description: 'Reject a draft email with an optional reason'
---

## Path Parameters

<ParamField path="id" type="integer" required>
  Activity ID
</ParamField>

## Body Parameters

<ParamField body="reason" type="string">
  Optional rejection reason for tracking
</ParamField>

<RequestExample>
```json
{
  "reason": "Email tone needs adjustment"
}
```
</RequestExample>

## Response

<ResponseField name="data" type="object">
  Rejected activity object
  
  <Expandable title="Rejected Activity Object">
    <ResponseField name="id" type="integer" required>
      Activity ID
    </ResponseField>
    
    <ResponseField name="sequence_lead_id" type="integer" required>
      Sequence lead ID
    </ResponseField>
    
    <ResponseField name="activity_type" type="string" required>
      Activity type
    </ResponseField>
    
    <ResponseField name="status" type="string" required>
      Status (will be "rejected")
    </ResponseField>
    
    <ResponseField name="subject" type="string">
      Email subject
    </ResponseField>
    
    <ResponseField name="due_at" type="string">
      Original scheduled time
    </ResponseField>
    
    <ResponseField name="rejected_at" type="string" required>
      Rejection timestamp
    </ResponseField>
    
    <ResponseField name="rejection_reason" type="string">
      Rejection reason
    </ResponseField>
  </Expandable>
</ResponseField>
