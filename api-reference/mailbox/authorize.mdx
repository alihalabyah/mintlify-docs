---
title: 'OAuth Authorization'
api: 'GET /auth/google'
description: 'Exchange OAuth authorization code for access tokens and create/update integration'
---

## Query Parameters

<ParamField query="authorizeGmail" type="boolean" required>
  Must be `true` to authorize Gmail
</ParamField>

<ParamField query="code" type="string" required>
  OAuth authorization code from Google
</ParamField>

<ParamField query="redirect_uri" type="string" required>
  OAuth redirect URI (must match the one registered with Google)
</ParamField>

## Response

<ResponseField name="data" type="object">
  <Expandable title="Integration Object">
    <ResponseField name="id" type="integer" required>
      Integration ID
    </ResponseField>
    
    <ResponseField name="user_id" type="integer" required>
      Owner user ID
    </ResponseField>
    
    <ResponseField name="type" type="string" required>
      Integration type ("gmail")
    </ResponseField>
    
    <ResponseField name="account_identifier" type="string" required>
      Email address from Google account
    </ResponseField>
    
    <ResponseField name="status" type="string" required>
      Connection status (will be "connected" on success)
    </ResponseField>
    
    <ResponseField name="authorized_gmail" type="boolean" required>
      Gmail authorization status (will be true)
    </ResponseField>
    
    <ResponseField name="per_day" type="integer" required>
      Daily email limit (default: 200)
    </ResponseField>
    
    <ResponseField name="per_hour" type="integer" required>
      Hourly email limit (default: 20)
    </ResponseField>
    
    <ResponseField name="min_delay_seconds" type="integer" required>
      Minimum delay between emails (default: 600)
    </ResponseField>
    
    <ResponseField name="metadata" type="object">
      Additional metadata
    </ResponseField>
    
    <ResponseField name="token_expiry_date" type="string">
      OAuth token expiry timestamp
    </ResponseField>
    
    <ResponseField name="created_at" type="string" required>
      Creation timestamp
    </ResponseField>
    
    <ResponseField name="updated_at" type="string" required>
      Last update timestamp
    </ResponseField>
  </Expandable>
</ResponseField>

<Note>
  If an integration with the same `user_id`, `type`, and `account_identifier` already exists, it will be updated. Otherwise, a new integration is created. The unique constraint prevents duplicate integrations.
</Note>
