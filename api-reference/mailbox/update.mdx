---
title: 'Update Integration Settings'
api: 'PATCH /integrations/:id'
description: 'Update rate limits, timezone, or status for a mailbox integration'
---

## Path Parameters

<ParamField path="id" type="integer" required>
  Integration ID
</ParamField>

## Body Parameters

<ParamField body="per_day" type="integer">
  Daily email limit (1-1000)
</ParamField>

<ParamField body="per_hour" type="integer">
  Hourly email limit
</ParamField>

<ParamField body="min_delay_seconds" type="integer">
  Minimum delay between emails in seconds
</ParamField>

<ParamField body="timezone" type="string">
  Timezone identifier (e.g., "Africa/Cairo", "America/New_York")
</ParamField>

<ParamField body="status" type="string">
  Integration status: "connected", "paused"
</ParamField>

<RequestExample>
```json
{
  "per_day": 200,
  "per_hour": 20,
  "min_delay_seconds": 600,
  "timezone": "Africa/Cairo",
  "status": "paused"
}
```
</RequestExample>
