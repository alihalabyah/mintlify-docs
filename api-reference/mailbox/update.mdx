---
title: 'Update Integration Settings'
api: 'PATCH /integrations/:id'
description: 'Update rate limits or status for a mailbox integration'
---

## Path Parameters

<ParamField path="id" type="integer" required>
  Integration ID
</ParamField>

## Body Parameters

<ParamField body="per_day" type="integer">
  Daily email limit (1-1000)
</ParamField>

<ParamField body="per_hour" type="integer">
  Hourly email limit
</ParamField>

<ParamField body="min_delay_seconds" type="integer">
  Minimum delay between emails in seconds
</ParamField>

<ParamField body="status" type="string">
  Integration status: `connected`, `disconnected`, `failed`
</ParamField>

<ParamField body="metadata" type="object">
  Additional metadata for the integration (JSON)
</ParamField>

<RequestExample>
```json
{
  "per_day": 200,
  "per_hour": 20,
  "min_delay_seconds": 600,
  "status": "connected"
}
```
</RequestExample>
