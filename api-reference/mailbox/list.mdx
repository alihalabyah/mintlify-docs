---
title: 'List Mailbox Integrations'
api: 'GET /integrations/mailboxes'
description: 'Get all mailbox integrations for the current user with their settings and status'
---

## Response

<ResponseField name="data" type="array">
  Array of mailbox integration objects
  
  <Expandable title="Integration Object">
    <ResponseField name="id" type="integer" required>
      Integration ID
    </ResponseField>
    
    <ResponseField name="user_id" type="integer" required>
      Owner user ID
    </ResponseField>
    
    <ResponseField name="type" type="string" required>
      Integration type (e.g., "gmail")
    </ResponseField>
    
    <ResponseField name="account_identifier" type="string" required>
      Email address or account identifier (unique per user+type)
    </ResponseField>
    
    <ResponseField name="status" type="string" required>
      Connection status: `connected`, `disconnected`, `failed`
    </ResponseField>
    
    <ResponseField name="per_day" type="integer" required>
      Daily email limit (rate limiting)
    </ResponseField>
    
    <ResponseField name="per_hour" type="integer" required>
      Hourly email limit (rate limiting)
    </ResponseField>
    
    <ResponseField name="min_delay_seconds" type="integer" required>
      Minimum delay between emails in seconds (rate limiting)
    </ResponseField>
    
    <ResponseField name="metadata" type="object">
      Additional metadata for the integration (JSON)
    </ResponseField>
    
    <ResponseField name="authorized_gmail" type="boolean" required>
      Whether Gmail OAuth is authorized
    </ResponseField>
    
    <ResponseField name="token_expiry_date" type="string">
      OAuth token expiry timestamp (ISO 8601)
    </ResponseField>
    
    <ResponseField name="created_at" type="string" required>
      Creation timestamp
    </ResponseField>
    
    <ResponseField name="updated_at" type="string" required>
      Last update timestamp
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="pagination" type="object">
  <Expandable title="Pagination Object">
    <ResponseField name="cursor" type="string">
      Next page cursor
    </ResponseField>
    
    <ResponseField name="has_more" type="boolean">
      Whether more results are available
    </ResponseField>
    
    <ResponseField name="count" type="integer">
      Total count of results
    </ResponseField>
  </Expandable>
</ResponseField>
