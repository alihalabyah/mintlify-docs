---
title: 'Update Sequence Lead'
api: 'PATCH /sequences/:sequenceId/leads/:leadId'
description: "Update a lead's introducers, status, or next follow-up time"
---

## Path Parameters

<ParamField path="sequenceId" type="integer" required>
  Sequence ID
</ParamField>

<ParamField path="leadId" type="integer" required>
  Lead ID
</ParamField>

## Body Parameters

<ParamField body="introducers" type="array">
  Array of introducer objects with lead_id and rank
  
  <Expandable title="Introducer Object">
    <ParamField body="lead_id" type="integer" required>
      Introducer lead ID
    </ParamField>
    
    <ParamField body="rank" type="integer" required>
      Priority rank (1 is highest priority)
    </ParamField>
  </Expandable>
</ParamField>

<ParamField body="status" type="string">
  Lead status: `active`, `paused`, `finished`
</ParamField>

<ParamField body="detailed_status" type="string">
  Detailed status: `in_progress`, `completed`, `not_a_fit`, `no_further_action`
</ParamField>

<ParamField body="next_followup_at" type="string">
  Next follow-up/reassessment timestamp (ISO 8601)
</ParamField>

<ParamField body="lead_data" type="object">
  Update lead information stored in JSON field
</ParamField>

<RequestExample>
```json
{
  "introducers": [
    {
      "lead_id": 742,
      "rank": 1
    },
    {
      "lead_id": 889,
      "rank": 2
    }
  ],
  "status": "active",
  "next_followup_at": "2025-11-15T10:00:00Z"
}
```
</RequestExample>
