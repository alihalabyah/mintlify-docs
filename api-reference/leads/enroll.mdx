---
title: 'Enroll Leads in Sequence'
api: 'POST /sequences/:id/leads'
description: 'Bulk enroll leads into a sequence'
---

## Path Parameters

<ParamField path="id" type="integer" required>
  Sequence ID
</ParamField>

## Body Parameters

<ParamField body="lead_ids" type="array" required>
  Array of lead IDs to enroll
</ParamField>

<ParamField body="integration_id" type="integer" required>
  Mailbox integration ID to use for sending emails
</ParamField>

<RequestExample>
```json
{
  "lead_ids": [501, 502, 503],
  "integration_id": 42
}
```
</RequestExample>

## Response

<ResponseField name="data" type="object">
  <Expandable title="Enrollment Result">
    <ResponseField name="sequence_id" type="string" required>
      Sequence ID
    </ResponseField>
    
    <ResponseField name="enrolled_count" type="integer" required>
      Number of leads successfully enrolled
    </ResponseField>
    
    <ResponseField name="items" type="array" required>
      Array of enrolled lead objects
    </ResponseField>
  </Expandable>
</ResponseField>
