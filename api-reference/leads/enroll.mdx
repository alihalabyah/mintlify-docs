---
title: 'Enroll Leads in Sequence'
api: 'POST /sequences/:sequenceId/leads'
description: 'Bulk enroll leads into a sequence with initial status set to active'
---

## Path Parameters

<ParamField path="sequenceId" type="integer" required>
  Sequence ID
</ParamField>

## Body Parameters

<ParamField body="lead_ids" type="array" required>
  Array of lead IDs to enroll
</ParamField>

<ParamField body="lead_data" type="object">
  Optional lead data to store (will be fetched from Postgres if not provided)
</ParamField>

<RequestExample>
```json
{
  "lead_ids": [501, 502, 503]
}
```
</RequestExample>

## Response

<ResponseField name="data" type="object">
  <Expandable title="Enrollment Result">
    <ResponseField name="sequence_id" type="integer" required>
      Sequence ID
    </ResponseField>
    
    <ResponseField name="enrolled_count" type="integer" required>
      Number of leads successfully enrolled
    </ResponseField>
    
    <ResponseField name="items" type="array" required>
      Array of enrolled lead objects
      
      <Expandable title="Enrolled Lead">
        <ResponseField name="id" type="integer" required>
          Sequence lead ID
        </ResponseField>
        
        <ResponseField name="lead_id" type="integer" required>
          Lead ID
        </ResponseField>
        
        <ResponseField name="status" type="string" required>
          Initial status (always `active`)
        </ResponseField>
        
        <ResponseField name="next_followup_at" type="string" required>
          Set to NOW() for immediate processing
        </ResponseField>
        
        <ResponseField name="lead_data" type="object">
          Lead information
        </ResponseField>
        
        <ResponseField name="created_at" type="string" required>
          Enrollment timestamp
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

<Note>
  After enrollment, leads have `status='active'` and `next_followup_at=NOW()` to trigger immediate AI planning. Introducers should be assigned via the Update Sequence Lead endpoint to set the lead to active status.
</Note>
