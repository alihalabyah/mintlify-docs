---
title: 'Get Lead Activities'
api: 'GET /sequences/:sequenceId/leads/:leadId/activities'
description: 'Get all activity history for a specific lead in a sequence'
---

## Path Parameters

<ParamField path="sequenceId" type="integer" required>
  Sequence ID
</ParamField>

<ParamField path="leadId" type="integer" required>
  Lead ID
</ParamField>

## Query Parameters

<ParamField query="activity_type" type="string">
  Filter by type: `incoming_email`, `outgoing_email`, `system_note`
</ParamField>

## Response

<ResponseField name="data" type="array">
  Array of activity objects ordered by created_at ASC
  
  <Expandable title="Activity Object">
    <ResponseField name="id" type="integer" required>
      Activity ID
    </ResponseField>
    
    <ResponseField name="sequence_lead_id" type="integer" required>
      Sequence lead ID
    </ResponseField>
    
    <ResponseField name="activity_type" type="string" required>
      Type: `incoming_email`, `outgoing_email`, `system_note`
    </ResponseField>
    
    <ResponseField name="is_approved" type="boolean">
      Approval status (for outgoing_email activities)
    </ResponseField>
    
    <ResponseField name="status" type="string">
      Status: `pending`, `approved`, `rejected`
    </ResponseField>
    
    <ResponseField name="due_at" type="string">
      Planned execution timestamp
    </ResponseField>
    
    <ResponseField name="executed_at" type="string">
      Actual execution timestamp
    </ResponseField>
    
    <ResponseField name="integration_thread_id" type="string">
      Gmail thread ID
    </ResponseField>
    
    <ResponseField name="integration_message_id" type="string">
      Gmail message ID
    </ResponseField>
    
    <ResponseField name="ai_confidence_score" type="number">
      AI confidence score (0.00-1.00)
    </ResponseField>
    
    <ResponseField name="ai_payload" type="object">
      Full AI context (JSON)
    </ResponseField>
    
    <ResponseField name="rejection_reason" type="string">
      User rejection reason
    </ResponseField>
    
    <ResponseField name="created_at" type="string" required>
      Creation timestamp
    </ResponseField>
  </Expandable>
</ResponseField>
