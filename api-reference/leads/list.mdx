---
title: 'List Sequence Leads'
api: 'GET /sequences/:id/leads'
description: 'List all leads enrolled in a sequence with their status and statistics'
---

## Path Parameters

<ParamField path="id" type="integer" required>
  Sequence ID
</ParamField>

## Query Parameters

<ParamField query="status" type="string">
  Filter by lead status: `active`, `paused`, `finished`
</ParamField>

<ParamField query="detailed_status" type="string">
  Filter by detailed status: `in_progress`, `completed`, `not_a_fit`, `no_further_action`
</ParamField>

## Response

<ResponseField name="data" type="array">
  Array of sequence lead objects
  
  <Expandable title="Sequence Lead Object">
    <ResponseField name="id" type="integer" required>
      Sequence lead ID
    </ResponseField>
    
    <ResponseField name="sequence_id" type="integer" required>
      Sequence ID
    </ResponseField>
    
    <ResponseField name="lead_id" type="integer" required>
      Lead ID (reference to leads/people table)
    </ResponseField>
    
    <ResponseField name="status" type="string" required>
      Current status: `active`, `paused`, `finished`
    </ResponseField>
    
    <ResponseField name="detailed_status" type="string">
      Detailed status: `in_progress`, `completed`, `not_a_fit`, `no_further_action`
    </ResponseField>
    
    <ResponseField name="next_followup_at" type="string">
      Next scheduled follow-up/reassessment timestamp
    </ResponseField>
    
    <ResponseField name="introducers" type="array">
      Array of introducer objects: `[{"lead_id":501,"rank":1}]`
    </ResponseField>
    
    <ResponseField name="lead_data" type="object">
      Lead information from Postgres (name, company, email, etc.)
    </ResponseField>
    
    <ResponseField name="created_at" type="string" required>
      Enrollment timestamp
    </ResponseField>
    
    <ResponseField name="updated_at" type="string" required>
      Last update timestamp
    </ResponseField>
  </Expandable>
</ResponseField>
