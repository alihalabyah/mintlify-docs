---
title: 'Remove Lead from Sequence'
api: 'DELETE /sequences/:sequenceId/leads/:leadId'
description: 'Remove a lead from a sequence by setting status to finished'
---

## Path Parameters

<ParamField path="sequenceId" type="integer" required>
  Sequence ID
</ParamField>

<ParamField path="leadId" type="integer" required>
  Lead ID
</ParamField>

## Response

<ResponseField name="data" type="object">
  Updated lead object with status set to finished
  
  <Expandable title="Removed Lead Object">
    <ResponseField name="id" type="integer" required>
      Sequence lead ID
    </ResponseField>
    
    <ResponseField name="sequence_id" type="integer" required>
      Sequence ID
    </ResponseField>
    
    <ResponseField name="lead_id" type="integer" required>
      Lead ID
    </ResponseField>
    
    <ResponseField name="status" type="string" required>
      Status (will be `finished`)
    </ResponseField>
    
    <ResponseField name="detailed_status" type="string">
      Detailed status (will be `no_further_action`)
    </ResponseField>
    
    <ResponseField name="introducers" type="array">
      Introducer list (preserved)
    </ResponseField>
    
    <ResponseField name="lead_data" type="object">
      Lead information
    </ResponseField>
    
    <ResponseField name="updated_at" type="string" required>
      Removal timestamp
    </ResponseField>
  </Expandable>
</ResponseField>

<Note>
  This is a soft delete. The record remains in the database with `status='finished'` and `detailed_status='no_further_action'`. No further automated actions will be taken for this lead.
</Note>
